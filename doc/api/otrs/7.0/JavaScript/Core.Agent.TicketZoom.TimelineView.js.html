"use strict";

var Core = Core || {};
Core.Agent = Core.Agent || {};
Core.Agent.TicketZoom = Core.Agent.TicketZoom || {};

/**
 * @namespace
 * @exports TargetNS as Core.Agent.TicketZoom
 * @description
 *      This namespace contains the special module functions for TicketZoom.
 */
Core.Agent.TicketZoom.TimelineView = (function (TargetNS) {

    /**
     * @public
     * @name Init
     * @memberof Core.Agent.TicketZoom.TimelineView
     * @param {Integer} TicketID
     * @param {Array} ListObjects
     * @param {Integer} ArticleID
     * @description
     *      Initializes the timeline view behavior.
     */
    TargetNS.Init = function(TicketID, ListObjects, ArticleID) {

        var LastItem = {},
            $LastItemObj,
            LastPosition = {},
            SimilarCounter = 1;

        // in case there are no events to display except for the NewTicket event,
        // show a fixed text item
        if (ListObjects.length > 1) {
            $('li.NoEvents').remove();
        }

        $.each(ListObjects, function(Index, Item) {
            var Margin = 0,
                $PrevObj,
                PrevItem = '',
                PrevPosition = {},
                IsConnectedItem,
                $ListItem = $(
                    Core.Template.Render('Agent/TicketZoom/TimelineView', {
                        Item: Item
                    })
                );

            // add consecutive items to the same list
            if (LastItem.time === Item.time) {
                Item.orientation = LastItem.orientation;
            }

            // check for similar items
            if (LastItem.type !== 'NewTicket' && !LastItem.article_id && LastItem.orientation && Item.orientation === LastItem.orientation && LastItem.type === Item.type && LastItem.time === Item.time) {
                SimilarCounter++;
                $LastItemObj.find('h3').find('em').text(SimilarCounter + 'x');
                $LastItemObj.find('p').append('<br>' + Item.name);
                return true;
            }
            else {
                SimilarCounter = 1;
            }

            // append item to the particular list
            $ListItem.appendTo($('.TimelineView ul.List' + Item.orientation));

            // check if we need to display the text fadeout overlay bar
            if ($ListItem.find('.PlainText').height() > 20) {
                $ListItem.find('.PlainText').append('<span class="FadeOut"></span>');
            }

            if (Item.type !== 'NewTicket') {
                if (Item.article_id) {
                    if (!$ListItem.hasClass('Seen')) {
                        $ListItem.find('div.UnreadArticles, .ShowFullArticleText').bind('click', function() {
                            Core.Agent.TicketZoom.MarkAsSeen(TicketID, Item.article_id, 0);
                        });
                    }

                    // check if the item has attachments
                    if ($('.ArticleAttachments.' + Item.id).length) {
                        $ListItem
                            .find('ul li.AttachmentsLink')
                            .bind('click', function () {
                                var Position = $(this).find('i').offset();
                                Core.UI.Dialog.ShowContentDialog($('.ArticleAttachments.ArticleID_' + Item.article_id).find('.AttachmentData').html(), 'Attachments', Position.top - $(window).scrollTop(), parseInt(Position.left, 10) + 25);
                                return false;
                            });
                    }
                }

                IsConnectedItem = false;

                // decide how much margin needs to be added
                if (LastItem.orientation && Item.orientation === LastItem.orientation) {

                    // if there are two consecutive items, just add fixed 15px top margin.
                    Margin = 15;

                    if (LastItem.time === Item.time && Item.type !== 'NewTicket') {
                        $ListItem.addClass('Connected');
                        IsConnectedItem = true;
                        Margin = 5;
                    }
                }
                else if (LastItem.orientation && Item.orientation !== LastItem.orientation) {

                    // if the previous item is on the other list, check the position of the previous item
                    // and add so much margin that the current item is really displayed after this item.

                    // case 1: current item is the first item in its parent list
                    //      substract fixed margin of list element
                    //      AND add the top margin of the preceding element
                    //      AND add the height of the title bar of the preceding element
                    Margin = LastPosition.top
                        - parseInt($('.TimelineView > ul').css('margin-top'), 10)
                        + parseInt($LastItemObj.css('margin-top'), 10)
                        + $LastItemObj.find('h3').outerHeight();

                    // we need to take into account the previous item for the following case. But if the previous item is
                    // the 'NewTicket' item, we need to skip this one and look on more step back, because the 'NewTicket'
                    // item is treated specially.
                    $PrevObj = $ListItem.prev('li:not(.NewTicket)');

                    // case 2: current item has at least one previous item in its parent list
                    if ($PrevObj.length) {

                        PrevItem = $PrevObj;
                        PrevPosition = PrevItem.position();

                        // substract the previous items position and height from the calculated margin, because
                        // we don't want to add the whole margin. If we would, we would end up having way too much
                        // margin added for each element, because we would not take into account the elements that
                        // precede our current element
                        Margin = Margin - (PrevPosition.top + $PrevObj.outerHeight(true)) + 20;

                        // margin can fall below 0 if the previous item in the
                        // current list is already taller than the margin that we need. In this
                        // case we just set it to 15 like always.
                        if (Margin < 15) {
                            Margin = 15;
                        }
                    }
                }

                $ListItem.css('margin-top', Margin);
            }

            // save current item and real display position for later comparison and offsetting
            if (Item.type !== 'NewTicket' && !IsConnectedItem) {
                LastItem = Item;
                $LastItemObj = $ListItem;
                LastPosition = $ListItem.position();
            }
        });

        $('.TimelineView').children('span').remove();

        // show overlay with full text when clicking the zoom icon
        $('.TimelineView').on('click', '.ArticleTitle, .ShowFullArticleText', function(Event) {

            var $HTMLObj,
                ArticleID = $(this).closest('li').attr('id').substring(10),
                DialogTitle = '',
                TimelineView = Core.Config.Get('TimelineView'),
                Article,
                Index;

            // Find article.
            for (Index = 0; Index < TimelineView.Data.Items.length; Index++) {
                if (TimelineView.Data.Items[Index].ArticleID === ArticleID) {
                    Article = TimelineView.Data.Items[Index];
                    break;
                }
            }

            if ($(Event.target).hasClass('fa-star')) {
                return false;
            }

            Event.stopImmediatePropagation();

            // init article content
            $HTMLObj = $(Article.ArticleData.ArticleHTML);
            DialogTitle = Article.ArticleData.ArticleFields.Subject.Value;

            Core.UI.Dialog.ShowContentDialog(
                $HTMLObj,
                DialogTitle,
                '20px',
                'Center',
                true
            );

            // open popup links correctly
            $('.ActionsTimelineView a.AsPopup').bind('click', function () {
                var Matches,
                    PopupType = 'TicketAction';

                Matches = $(this).attr('class').match(/PopupType_(\w+)/);
                if (Matches) {
                    PopupType = Matches[1];
                }
                Core.UI.Popup.OpenPopup($(this).attr('href'), PopupType);
                return false;
            });

            $(".ActionsTimelineView .Actions select").each(function() {
                Core.Agent.TicketZoom.ArticleActionMenuDropdown(
                    $(this).closest("form").attr("id"),
                    $(this).attr("name")
                );
            });

            $('a.SplitSelection').unbind('click.SplitSelection').bind('click.SplitSelection', function() {
                Core.Agent.TicketSplit.OpenSplitSelection($(this).attr('href'));
                return false;
            });

            return false;
        });

        // If we have an article id and config NoTimelineViewAutoArticle is diabled, go directly to this article.
        if (ArticleID && Core.Config.Get('NoTimelineViewAutoArticle') !== '1') {
            $('.TimelineView li#ArticleID_' + ArticleID).find('.ShowFullArticleText').click();
        }
    };

    /**
     * @public
     * @name InitTimelineView
     * @memberof Core.Agent.TicketZoom
     * @function
     * @param {Object} TimelineView - data needed for initialization timeline view
     * @description
     *      This function initializes timeline view.
     */
    TargetNS.InitTimelineView = function(TimelineView) {
        var Index,
            ArticleID,
            ListObject = {},
            ListObjects = [];

        for (Index in TimelineView.Data.Items) {
            ArticleID = TimelineView.Data.Items[Index].ArticleID;

            ListObject = {
                orientation    : TimelineView.Data.Items[Index].Orientation,
                order          : TimelineView.Data.Items[Index].Counter,
                id             : (typeof ArticleID !== 'undefined' && ArticleID !== '') ? ('ArticleID_' + ArticleID) : '',
                class          : TimelineView.Data.Items[Index].Class,
                time           : TimelineView.Data.Items[Index].CreateTime,
                time_long      : TimelineView.Data.Items[Index].TimeLong,
                type           : TimelineView.Data.Items[Index].HistoryType,
                type_readable  : Core.Language.Translate(TimelineView.Data.Items[Index].HistoryTypeReadable),
                name           : TimelineView.Data.Items[Index].Name,
                article_id     : ArticleID,
                is_chat        : TimelineView.Data.Items[Index].IsChatArticle,
                article_data   : {},
                has_attachment :TimelineView.Data.Items[Index].HasAttachment
            };

            if (typeof ArticleID !== 'undefined' && ArticleID !== '') {
                ListObject.article_data = {
                    sender_type   : TimelineView.Data.Items[Index].ArticleData.SenderType,
                    is_visible_for_customer  : TimelineView.Data.Items[Index].ArticleData.IsVisibleForCustomer,
                    article_fields : $.map(TimelineView.Data.Items[Index].ArticleData.ArticleFields, function (Value) { return Value; }),
                    article_meta_fields : TimelineView.Data.Items[Index].ArticleData.ArticleMetaFields,
                    is_important  : TimelineView.Data.Items[Index].ArticleData.ArticleIsImportant,
                    is_seen       : TimelineView.Data.Items[Index].ArticleData.ArticleIsSeen,
                };

                ListObject.article_data.plain = TimelineView.Data.Items[Index].ArticleData.ArticlePlain.substring(0, 650);
                ListObject.article_data.html = TimelineView.Data.Items[Index].ArticleData.ArticleHTML;

                if (ListObject.article_data.html === undefined || ListObject.article_data.html === '') {
                    ListObject.article_data.html = TimelineView.Data.Items[Index].ArticleData.ArticlePlain;
                }
            }

            ListObjects.push(ListObject);
        }

        TargetNS.Init(TimelineView.Data.TicketID, ListObjects, TimelineView.Data.CurrentArticleID);
    }

    return TargetNS;
}(Core.Agent.TicketZoom.TimelineView || {}));
