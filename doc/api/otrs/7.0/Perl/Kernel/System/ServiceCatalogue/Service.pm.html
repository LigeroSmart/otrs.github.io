NAME

Kernel::System::ServiceCatalogue::Service

DESCRIPTION

All Service Catalogue Service functions.

PUBLIC INTERFACE

BUILD()

Customization of object construction, do not use it directly.

ServiceAdd()

Adds new Service in Service Catalogue.

    my $Success = $ServiceObject->ServiceAdd(
        InternalTitle => 'Service',               # (required) Service title
        ValidID       => 1,                       # (required) ValidID
        UserID        => 1,                       # (required) UserID
    );

ServiceGet()

Get Service data.

    my %Service = $ServiceObject->ServiceGet(
        ServiceID     => 1,             # (required) Service ID
                                        # or
        InternalTitle => 'service',     # (required) Internal Title
    );

Returns:

    %Service = (
        ServiceID     => 1,
        InternalTitle => 'Service name',
        UsageCounter  => 0,
        ValidID       => 1,
        CreateBy      => 1,
        CreateTime    => '2018-07-20 12:00:00',
        ChangeBy      => 1,
        ChangeTime    => '2018-07-20 12:00:00',
        Content       => {
            'de' => {
                ServiceContentID => 123,
                Title            => 'title',
                Text             => 'text',
                Link             => 'www.otrs.com',
                CreateTime       => '2018-07-20 12:00:00',
                CreateBy         => 1,
                ChangeTime       => '2018-07-20 12:00:00',
                ChangeBy         => 1,
                CategoryIDs      => [1, 2, 3],
            },
        },
    );

ServiceList()

Lists Services in Service Catalogue.

    my @Services = $ServiceObject->ServiceList(
        Valid      => 1,            # (optional) List only valid Services
        CategoryID => 123,          # (optional)

ServiceSearch()

Search for Services.

    my @Services = $ServiceObject->ServiceSearch(
        Language   => 'en',             # (required) Language
        CategoryID => 123,              # (optional) CategoryID
        Valid      => 1,                # (optional) Return only valid services
    );

Returns:

    @Services = (
        {
            ServiceID    => 123,
            Title        => 'Title in english',
            Text         => 'English content',
            Link         => 'www.otrs.com',
            CategoryID   => 123,
        },
        ...
    );

ServiceDelete()

Delete Service and Service content from the Service Catalogue.

    my $Success = $ServiceObject->ServiceDelete(
        ServiceID => 123,               # (required) ServiceID
    );

ServiceUpdate()

Update Service in Service Catalogue.

    my $Success = $ServiceObject->ServiceUpdate(
        ServiceID     => 1,                   # (required) Service ID to update
        InternalTitle => 'Category name',     # (required) Internal title
        ValidID     => 2,                     # (required) ValidID
        UserID      => 1,                     # (required) UserID
    );

ServiceUsageCounterIncrease()

Increase usage counter for provided ServiceID.

    my $Success = $ServiceObject->ServiceUsageCounterIncrease(
        ServiceID     => 1,        # (required) Service ID to update
    );

ServiceContentAdd()

Adds Service content.

    my $ServiceContentID = $ServiceObject->ServiceContentAdd(
        ServiceID     => 123,                     # (required) ServiceID
        Title         => 'Service',               # (required) Service title
        Text          => 'Text',                  # (required) Service text
        Language      => 'en',                    # (required) Service language
        UserID        => 1,                       # (required) UserID
        Link          => 'www.otrs.com',          # (required) Link
    );

Result: $ServiceContentID = 123;

ServiceContentGet()

Get Service content data.

    my %ServiceContent = $ServiceObject->ServiceContentGet(
        ServiceContentID => 123,        # (required) ServiceContentID
    );

Result:

    %ServiceContent = (
        ServiceContentID => 123
        ServiceID        => 245,
        Title            => 'Service',
        Text             => 'Text',
        Language         => 'en',
        Link             => 'www.otrs.com',
        CreateBy         => 1,
        CreateTime       => '2018-07-20 12:00:00',
        ChangeBy         => 1,
        ChangeTime       => '2018-07-20 12:00:00',
        CategoryIDs      => [1, 2, 3],
    );

ServiceContentUpdate()

Updates Service content.

    my $Success = $ServiceObject->ServiceContentUpdate(
        ServiceContentID => 123,                     # (required) ServiceContentID
        ServiceID        => 123,                     # (required) ServiceID (can't be updated)
        Title            => 'Service',               # (required) Service title
        Text             => 'Text',                  # (required) Service text
        UserID           => 1,                       # (required) UserID
        Link             => 'www.otrs.com',          # (required) Link
    );

Events: ServiceContentUpdate

ServiceContentDelete()

Delete Service content from the Service Catalogue.

    my $Success = $ServiceObject->ServiceContentDelete(
        ServiceContentID => 123,            # (required) ServiceContentID
        ServiceID        => 123,            # (required) ServiceID
    );

Events: ServiceContentDelete

ServiceContentCategorySet()

Set Service Categories.

    my $Success = $ServiceObject->ServiceContentCategorySet(
        ServiceContentID => 123,            # (required) ServiceContentID
        CategoryIDs      => [               # (required) List of Category IDs
            123,
            124,
            ...
        ],
        UserID           => 1,              # (required) UserID
    );

Events: ServiceContentCategorySet
